teamValueA = int(input('A : '))
teamValueB = int(input('B : '))
teamValueC = int(input('C : '))
running = 1
loopCount = 0

while running:
    functionF1 = abs(teamValueA - teamValueB)
    functionF2 = abs(teamValueA - teamValueC)
    functionF3 = abs(teamValueB - teamValueC)

    print('\nF: ', functionF1, functionF2, functionF3, '\nV: ', teamValueA, teamValueB, teamValueC)

    functionSum = functionF1 + functionF2 + functionF3
    functionMax = max(functionF1,functionF2,functionF3)

    print('FSum: ', functionSum, 'FMax: ', functionMax)

    if functionSum >= 3:
        loopCount += 1
        print('Sum is greater than three. Team balancing is required.')
        functionCalcF1 = abs(functionF1 - functionMax)
        functionCalcF2 = abs(functionF2 - functionMax)
        functionCalcF3 = abs(functionF3 - functionMax)

        if functionCalcF1 == 0:
            functionCalcF1 = teamValueA - teamValueB
            if functionCalcF1 > 0:
                teamValueA -= 1
                teamValueB += 1
            if functionCalcF1 < 0:
                teamValueB -= 1
                teamValueA += 1

            functionCalcF1 = -1
            functionCalcF2 = -1
            functionCalcF3 = -1

        if functionCalcF2 == 0:
            functionCalcF2 = teamValueA - teamValueC
            if functionCalcF2 > 0:
                teamValueA -= 1
                teamValueC += 1
            if functionCalcF2 < 0:
                teamValueC -= 1
                teamValueA += 1

            functionCalcF1 = -1
            functionCalcF2 = -1
            functionCalcF3 = -1

        if functionCalcF3 == 0:
            functionCalcF3 = teamValueB - teamValueC
            if functionCalcF3 > 0:
                teamValueB -= 1
                teamValueC += 1
            if functionCalcF3 < 0:
                teamValueC -= 1
                teamValueB += 1

            functionCalcF1 = -1
            functionCalcF2 = -1
            functionCalcF3 = -1
    else:
        print('Sum is less than three')
        running = 0

print('V: ', teamValueA, teamValueB, teamValueC)
print('Loop Count: ', loopCount)
